# йј ж ҮзӮ№еҮ»дәӢд»¶

button1.setOnMouseClicked(new EventHandler<MouseEvent>() {  
    @Override  
    public void handle(MouseEvent event) {  
        // иҺ·еҸ–зӣёеҜ№дәҺеңәжҷҜзҡ„йј ж ҮжҢҮй’Ҳеқҗж Ү  
        System.out.println("event.getSceneX() = " + event.getSceneX());  
        System.out.println("event.getSceneX() = " + event.getSceneX());  
        // иҺ·еҸ–зӣёеҜ№дәҺж•ҙдёӘеұҸе№•зҡ„йј ж ҮжҢҮй’Ҳеқҗж Ү  
        System.out.println("event.getScreenX() = " + event.getScreenX());  
        System.out.println("event.getScreenY() = " + event.getScreenY());  
        // иҺ·еҸ–зӣёеҜ№дәҺиҜҘз»„д»¶еҶ…йғЁзҡ„йј ж ҮжҢҮй’Ҳеқҗж Ү  
        System.out.println("event.getX() = " + event.getX());  
        System.out.println("event.getY() = " + event.getY());  
        // дәӢд»¶жәҗпјҲдәӢд»¶еҸ‘иө·иҖ…пјү  
        System.out.println("event.getSource() = " + event.getSource());  
        // дәӢд»¶зӣ®ж ҮпјҲзӮ№еҮ»зҡ„еҢәеҹҹпјҢдёҖиҲ¬жҳҜдәӢд»¶жәҗжҲ–иҖ…дәӢд»¶жәҗеҶ…йғЁзҡ„з»„д»¶пјү  
        System.out.println("event.getTarget() = " + event.getTarget());  
        // дәӢд»¶зұ»еһӢ  
        System.out.println("event.getEventType() = " + event.getEventType());  
        // иҺ·еҸ–зӮ№еҮ»зҡ„жҢүй”®  
        System.out.println("event.getButton() = " + event.getButton());  
        // жҳҜеҗҰжҢүдёӢCTRL  
        System.out.println("event.isControlDown() = " + event.isControlDown());  
        // иҝһз»ӯзӮ№еҮ»ж¬Ўж•°  
        System.out.println("event.getClickCount() = " + event.getClickCount());  
        // йј ж ҮеҸій”®жҳҜеҗҰжҢүдёӢпјҲе…Ҳй•ҝжҢүйј ж ҮеҸій”®пјҢз„¶еҗҺеҗҢж—¶еҚ•еҮ»йј ж Үе·Ұй”®пјҢжү“еҚ°trueпјү  
        System.out.println("йј ж ҮеҸій”®жҳҜеҗҰжҢүдёӢ = " + event.isSecondaryButtonDown());  
        if (event.getClickCount() == 2 && event.getButton() == MouseButton.PRIMARY) {  
            System.out.println("еҸҢеҮ»йј ж Үе·Ұй”®");  
        }  
    }  
});
